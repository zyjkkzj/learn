					2.设定生成图片大小生成二维码方法
					$("#code").qrcode({

					3.生成中文二维码的方法
					//	将中文字符串转意
					function toUtf8(str) {    
					    var out, i, len, c;    
					    out = "";    
					    len = str.length;    
					    for(i = 0; i < len; i++) {    
					        c = str.charCodeAt(i);    
					        if ((c >= 0x0001) && (c <= 0x007F)) {    
					            out += str.charAt(i);    
					        } else if (c > 0x07FF) {    
					            out += String.fromCharCode(0xE0 | ((c >> 12) & 0x0F));    
					            out += String.fromCharCode(0x80 | ((c >>  6) & 0x3F));    
					            out += String.fromCharCode(0x80 | ((c >>  0) & 0x3F));    
					        } else {    
					            out += String.fromCharCode(0xC0 | ((c >>  6) & 0x1F));    
					            out += String.fromCharCode(0x80 | ((c >>  0) & 0x3F));    
					        }    
					    }    
					    return out;    
					}

					//	定义字符串
					var str = toUtf8("看什么看，有啥好看的"); 

					//	生成二维码
					$('#code').qrcode({
						render: "table", //table方式 
					    width: 320, //宽度 
					    height: 320, //高度 
					    text: str //定义的字符串
					 });
				</script>
